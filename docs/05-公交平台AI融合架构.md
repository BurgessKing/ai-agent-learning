**Java 服务层（Spring Boot）→ Python Agent 调用**：

```java
/**
 * 调度服务 — Spring Boot 侧
 * 负责业务逻辑 + 调用 AI Agent 获取推荐
 */
@Service
public class DispatchService {
    
    @Autowired
    private VehicleRepository vehicleRepo;
    @Autowired
    private DriverRepository driverRepo;
    @Autowired
    private AiAgentClient aiAgentClient;  // 新增：调用 Python Agent
    
    /**
     * 获取 AI 调度推荐
     * 
     * 调用链: PC端 → DispatchController → DispatchService → AiAgentClient → Python Agent
     */
    public DispatchRecommendation getAiRecommendation(DispatchRequest request) {
        
        // 1. 从数据库查询候选资源（Java 擅长）
        List<Vehicle> availableVehicles = vehicleRepo.findAvailable(
            request.getDate(), request.getVehicleType(), request.getSeatCount());
        
        List<Driver> availableDrivers = driverRepo.findAvailable(
            request.getDate(), request.getMinExperience());
        
        // 2. 查询历史匹配记录和评分
        Map<String, MatchHistory> histories = getMatchHistories(
            request.getRouteId(), availableVehicles, availableDrivers);
        
        // 3. 查询约束条件（保养计划、司机偏好等）
        List<Constraint> constraints = getConstraints(
            availableVehicles, availableDrivers, request.getDate());
        
        // 4. 组装上下文，调用 AI Agent（Python）
        AgentContext context = AgentContext.builder()
            .orderInfo(request)
            .candidates(AiCandidate.builder()
                .vehicles(availableVehicles)
                .drivers(availableDrivers)
                .build())
            .matchHistories(histories)
            .constraints(constraints)
            .build();
        
        // 调用 Python Agent API
        AiResponse aiResponse = aiAgentClient.getDispatchRecommendation(context);
        
        // 5. 封装返回（Agent 推荐 + 原始数据，前端可以对比展示）
        return DispatchRecommendation.builder()
            .recommendation(aiResponse.getRecommendation())
            .alternatives(aiResponse.getAlternatives())
            .riskWarnings(aiResponse.getRiskWarnings())
            .confidence(aiResponse.getConfidence())
            .rawData(context)  // 前端可以"查看详情"
            .build();
    }
    
    /**
     * 执行调度（含冲突检测 + 分布式锁）
     */
    @Transactional
    public DispatchResult confirmDispatch(DispatchConfirmRequest confirm) {
        String lockKey = "dispatch:vehicle:" + confirm.getVehicleId();
        
        // 分布式锁防止并发冲突
        RLock lock = redissonClient.getLock(lockKey);
        try {
            if (!lock.tryLock(5, 30, TimeUnit.SECONDS)) {
                throw new DispatchConflictException("车辆正在被其他调度员操作");
            }
            
            // 乐观锁更新
            int rows = vehicleRepo.assignWithOptimisticLock(
                confirm.getVehicleId(), confirm.getRouteId());
            if (rows == 0) {
                throw new DispatchConflictException("车辆已被分配");
            }
            
            // 记录调度日志（用于 Agent 学习）
            dispatchLogRepo.save(DispatchLog.builder()
                .vehicleId(confirm.getVehicleId())
                .driverId(confirm.getDriverId())
                .routeId(confirm.getRouteId())
                .aiRecommended(confirm.isAiRecommended())
                .aiConfidence(confirm.getAiConfidence())
                .operatorId(confirm.getOperatorId())
                .build());
            
            return DispatchResult.success();
            
        } finally {
            if (lock.isHeldByCurrentThread()) {
                lock.unlock();
            }
        }
    }
}
```

**Python Agent 侧（调度 Agent 核心）**：

```python
"""
调度 Agent — Python/FastAPI 侧
接收 Java 侧传来的上下文，返回智能推荐
"""

from langchain.agents import AgentExecutor, create_openai_tools_agent
from langchain.tools import tool
from langchain.prompts import ChatPromptTemplate, MessagesPlaceholder
from pydantic import BaseModel
from typing import List, Optional
import json

# ==================== 输入模型 ====================

class CandidateVehicle(BaseModel):
    vehicle_id: str
    plate_no: str
    vehicle_type: str  # 大巴/中巴/商务车
    seats: int
    age_years: float
    mileage_km: int
    last_maintenance_date: str

class CandidateDriver(BaseModel):
    driver_id: str
    name: str
    experience_years: int
    rating: float  # 1-5
    home_address: str
    current_assignments: List[dict]

class MatchHistory(BaseModel):
    vehicle_id: str
    driver_id: str
    match_count: int
    avg_rating: float

class DispatchRequest(BaseModel):
    route_name: str
    order_type: str  # 长保/散包
    date: str
    time_slot: str
    passenger_count: int
    required_seats: int
    special_requirements: List[str]  # 如 ["需行李舱", "需轮椅通道"]

class AgentContext(BaseModel):
    order: DispatchRequest
    vehicles: List[CandidateVehicle]
    drivers: List[CandidateDriver]
    match_histories: List[MatchHistory]
    constraints: List[str]

# ==================== Agent Prompt ====================

DISPATCH_SYSTEM_PROMPT = """你是车辆调度优化专家，服务于公交出行平台。

## 你的职责
根据订单需求、可用资源和历史数据，推荐最优的车辆-司机匹配方案。

## 匹配原则（优先级从高到低）
1. **安全合规**: 司机驾龄、车辆年限必须满足要求
2. **匹配度**: 历史配合记录好的车-司机组合优先
3. **效率**: 司机通勤距离短、车辆利用率高
4. **成本**: 同等条件下选择油耗低、维护成本低的车辆

## 输出格式
你的最终回答必须包含：
1. **推荐方案**: 最推荐的车-司机组合，附理由
2. **备选方案**: 1-2个备选组合
3. **风险提示**: 任何需要注意的问题（如司机当天有其他排班、车辆即将保养等）
4. **匹配度评分**: 0-100分

## 注意
- 如果你发现合适的组合，明确说明为什么
- 如果没有完全合适的，说明差距在哪里，给出最接近的方案
"""

# ==================== Agent 实现 ====================

class DispatchAgent:
    """调度推荐 Agent"""
    
    def __init__(self, llm, tools):
        self.llm = llm
        self.tools = tools
        
        self.prompt = ChatPromptTemplate.from_messages([
            ("system", DISPATCH_SYSTEM_PROMPT),
            MessagesPlaceholder(variable_name="chat_history"),
            ("human", "{input}"),
            MessagesPlaceholder(variable_name="agent_scratchpad"),
        ])
        
        agent = create_openai_tools_agent(llm, tools, self.prompt)
        self.executor = AgentExecutor(
            agent=agent,
            tools=tools,
            verbose=True,
            max_iterations=8,
            handle_parsing_errors=True,
        )
    
    async def recommend(self, context: AgentContext) -> dict:
        """生成调度推荐"""
        
        # 构建结构化的输入
        input_text = f"""
## 订单信息
- 线路: {context.order.route_name}
- 类型: {context.order.order_type}
- 日期: {context.order.date}
- 时段: {context.order.time_slot}
- 人数: {context.order.passenger_count}人 (需{context.order.required_seats}座以上)
- 特殊要求: {context.order.special_requirements or '无'}

## 可用车辆 ({len(context.vehicles)}辆)
{json.dumps([v.model_dump() for v in context.vehicles], ensure_ascii=False, indent=2)}

## 可用司机 ({len(context.drivers)}名)
{json.dumps([d.model_dump() for d in context.drivers], ensure_ascii=False, indent=2)}

## 历史匹配记录
{json.dumps([h.model_dump() for h in context.match_histories], ensure_ascii=False, indent=2)}

## 约束条件
{chr(10).join(f'- {c}' for c in context.constraints)}

请分析并推荐最优的车-司机匹配方案。
"""
        
        result = await self.executor.ainvoke({"input": input_text})
        
        # 解析 Agent 输出，提取结构化字段
        output = result["output"]
        
        return {
            "recommendation": self._extract_section(output, "推荐方案"),
            "alternatives": self._extract_section(output, "备选方案"),
            "risk_warnings": self._extract_section(output, "风险提示"),
            "score_breakdown": self._extract_score(output),
            "full_response": output,
            "intermediate_steps": result.get("intermediate_steps", []),
        }
    
    def _extract_section(self, text: str, section: str) -> str:
        """从 Agent 输出中提取指定段落"""
        import re
        pattern = rf'#+\s*{section}[:\s]*(.+?)(?=#+\s|\Z)'
        match = re.search(pattern, text, re.DOTALL)
        return match.group(1).strip() if match else ""

    def _extract_score(self, text: str) -> int:
        import re
        match = re.search(r'(\d+)\s*分', text)
        return int(match.group(1)) if match else 0

# ==================== FastAPI 接口 ====================

from fastapi import FastAPI

dispatch_app = FastAPI(title="调度 Agent API")

@dispatch_app.post("/api/dispatch/recommend")
async def get_dispatch_recommendation(context: AgentContext):
    """调度推荐接口 — 被 Java DispatchService 调用"""
    agent = get_dispatch_agent()  # 单例获取
    result = await agent.recommend(context)
    return result
```

```java
@Service
public class PricingService {
    
    @Autowired
    private PriceRuleEngine ruleEngine;  // 现有的规则引擎
    @Autowired
    private AiAgentClient aiAgentClient;
    @Autowired
    private PriceHistoryRepository historyRepo;
    
    /**
     * 获取 AI 增强报价
     */
    public QuoteResponse getAiQuote(QuoteRequest request) {
        
        // 1. 规则引擎精确计算（Java 负责精确计算部分）
        PriceBreakdown breakdown = ruleEngine.calculate(
            request.getDistance(),
            request.getPassengerCount(),
            request.getVehicleType(),
            request.getDate(),
            request.getAdditionalServices()
        );
        
        // 2. 查询历史报价和竞品价格
        List<PriceHistory> history = historyRepo.findSimilar(
            request.getDestination(), request.getPassengerCount());
        MarketReference marketRef = getMarketReference(request.getDestination());
        
        // 3. 调用 AI Agent 获取策略建议
        AiQuoteContext aiContext = AiQuoteContext.builder()
            .request(request)
            .priceBreakdown(breakdown)
            .priceHistory(history)
            .marketReference(marketRef)
            .customerProfile(getCustomerProfile(request.getCustomerId()))
            .build();
        
        AiQuoteResponse aiResponse = aiAgentClient.getQuoteStrategy(aiContext);
        
        // 4. 组装返回
        return QuoteResponse.builder()
            .standardPrice(breakdown.getTotal())
            .recommendedQuote(aiResponse.getRecommendedPrice())
            .alternatives(aiResponse.getAlternatives())
            .negotiationTips(aiResponse.getNegotiationTips())
            .riskWarnings(aiResponse.getRiskWarnings())
            .priceBreakdown(breakdown)
            .build();
    }
}
```

### D6：项目沉淀 — 构建 Agent 系统的经验总结

```
═══════════════════════════════════════════════════════════
        从公交出行平台 Agent 化中沉淀的实战经验
═══════════════════════════════════════════════════════════

1. 架构原则:
   ✅ LLM 不做精确计算 → 规则引擎 + Agent 策略层分离
   ✅ Agent 不直接操作数据库 → 通过 API 调用业务服务
   ✅ Agent 输出需校验 → 关键操作前有确认环节 (Human-in-the-loop)
   ✅ 合理降级 → Agent 不可用时，回退到传统手动流程

2. 成本优化:
   ✅ 缓存常见问题的 Agent 响应 (减少 LLM 调用)
   ✅ 简单任务用小模型 (如需50座大巴→一次API查询足够)
   ✅ 批量调度在非高峰期执行 (节省紧急调度的 Token 消耗)

3. 质量保障:
   ✅ 调度推荐准确率: 人工采纳率 > 85%
   ✅ 报价合理性: 不能低于成本价 (代码硬约束)
   ✅ A/B 测试: 灰度发布，对比 Agent 辅助前后的调度效率

4. 与传统系统的集成模式:
   ✅ Java 负责: 事务、并发、数据一致性、精确计算
   ✅ Python Agent 负责: 推理、推荐、策略分析
   ✅ 通信方式: HTTP REST (简单场景) / gRPC (高性能场景)
```
